\newcommand{\makeaxis}[2]{
    \begin{minipage}{0.4\paperwidth}
        \begin{tikzpicture}
            \begin{axis} [
                title={#1}
            ]
                \addplot+ [
                    scatter,
                    only marks,
                    scatter src={\thisrow{freq}/\thisrow{max_freq}}
                    ] table [x=n,y=value] {#2};
            \end{axis}
        \end{tikzpicture}
    \end{minipage}
}
